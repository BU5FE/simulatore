Cliente: ${utente} Data: ${new Date().toLocaleDateString()}

`; let totalSaveAnnuo = 0; // --- LOGICA LUCE --- if (document.getElementById('utilityType').value.includes('light')) { const freq = parseInt(document.getElementById('freqLuce').value); const consAnnuo = parseFloat(document.getElementById('annuoLuce').value); const spesaPeriodo = parseFloat(document.getElementById('costMateriaLuce').value); const mKey = document.getElementById('monthLuce').value; const consPeriodo = parseFloat(document.getElementById('kWhTot').value) || 0; let spreadL = 0.055; // Default media let pcvMensile = (userType === 'consumer' ? 9 : 12.5); // Calcolo nuova spesa nel periodo (PUN + Spread + PCV) let costoEnergiaUnitario = DB_PRICES.pun[mKey] + spreadL; let nuovaSpesaMateriaLuce = (consPeriodo * costoEnergiaUnitario) + (pcvMensile * freq); let risparmioPeriodoLuce = spesaPeriodo - nuovaSpesaMateriaLuce; let saveAnnuoLuce = (risparmioPeriodoLuce / (consPeriodo || 1)) * consAnnuo; totalSaveAnnuo += saveAnnuoLuce; reportHtml += `
